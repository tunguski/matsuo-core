22  /**
23   * Organization unit controller.
24   *
25   * @author Marek Romanowski
26   * @since Jul 18, 2013
27   */
28  @RestController
29  @RequestMapping("/organizationUnits")
30  public class OrganizationUnitController extends AbstractSimpleController<OrganizationUnit> {
31  
32  
33    @Override
34    protected List<String> queryMatchers() {
35      return asList("fullName");
36    }
37  
38  
39    @Override
40    protected List<? extends Initializer<OrganizationUnit>> entityInitializers() {
41      return asList(new CompanyInitializer());
42    }
43  
44  
45    @Override
46    @RequestMapping(value = "/{id}", method = GET)
47    public HttpEntity<OrganizationUnit> find(@PathVariable("id") Integer id) {
48      HttpEntity<OrganizationUnit> entity = super.find(id);
49      entity.getBody().getEmployees().size();
50      return entity;
51    }
52  
53  
54    @RequestMapping(value = "/{id}/employee/{idEmployee}", method = POST)
55    @ResponseStatus(OK)
56    public Person addEmployee(@PathVariable("id") Integer id,
57                              @PathVariable("idEmployee") Integer idEmployee) {
58      Person person = database.findById(Person.class, idEmployee);
59      OrganizationUnit organizationUnit = database.findById(OrganizationUnit.class, id, entityInitializers);
60  
61      organizationUnit.getEmployees().add(person);
62      database.update(organizationUnit);
63  
64      return person;
65    }
66  
67  
68    @RequestMapping(value = "/{id}/employee/{idEmployee}", method = DELETE)
69    @ResponseStatus(OK)
70    public void removeEmployee(@PathVariable("id") Integer id,
71                               @PathVariable("idEmployee") Integer idEmployee) {
72      OrganizationUnit organizationUnit = database.findById(OrganizationUnit.class, id, entityInitializers);
73  
74      Iterator<Person> iterator = organizationUnit.getEmployees().iterator();
75      while (iterator.hasNext()) {
76        Person person = iterator.next();
77        if (person.getId().equals(idEmployee)) {
78          iterator.remove();
79        }
80      }
81  
82      database.update(organizationUnit);
83    }
84  }
