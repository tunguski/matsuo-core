3   import org.springframework.http.ResponseEntity;
4   import org.springframework.validation.BindingResult;
5   import org.springframework.validation.FieldError;
6   import org.springframework.validation.ObjectError;
7   import org.springframework.web.bind.MethodArgumentNotValidException;
8   import org.springframework.web.bind.annotation.ControllerAdvice;
9   import org.springframework.web.bind.annotation.ExceptionHandler;
10  import pl.matsuo.core.validation.ValidationErrors;
11  
12  import static org.springframework.http.HttpStatus.*;
13  
14  
15  @ControllerAdvice
16  public class ValidationExceptionHandler {
17  
18  
19    @ExceptionHandler({MethodArgumentNotValidException.class})
20    public ResponseEntity<Object> handleValidationException(MethodArgumentNotValidException pe) {
21      BindingResult bindingResult = pe.getBindingResult();
22      ValidationErrors errors = new ValidationErrors();
23  
24      for (FieldError fieldError : bindingResult.getFieldErrors()) {
25        errors.getFieldErrors().put(fieldError.getField(), fieldError.getCode());
26      }
27  
28      for (ObjectError error : bindingResult.getGlobalErrors()) {
29        errors.getGlobalErrors().add(error.getCode());
30      }
31  
32      return new ResponseEntity<Object>(errors, BAD_REQUEST);
33    }
34  }
