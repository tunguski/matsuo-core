18  public class TestDataDiscoveryRegisteringBeanFactoryPostProcessor  implements BeanFactoryPostProcessor {
19  
20  
21    public void discover(Class<?> clazz, Set<Class> toAdd) {
22      DiscoverTypes annotation = clazz.getAnnotation(DiscoverTypes.class);
23      if (annotation != null) {
24        toAdd.addAll(Arrays.asList(annotation.value()));
25        for (Class nextClass : annotation.value()) {
26          discover(nextClass, toAdd);
27        }
28      }
29    }
30  
31  
32    public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) {
33      try {
34        BeanDefinitionRegistry factory = (BeanDefinitionRegistry) beanFactory;
35        Set<Class> toAdd = new HashSet<>();
36  
37        // odnajduje wszystkie referencje opisane przez Discover
38        for (String beanName : beanFactory.getBeanDefinitionNames()) {
39          BeanDefinition beanDefinition = beanFactory.getBeanDefinition(beanName);
40          if (beanDefinition.getBeanClassName() != null) {
41            discover(Class.forName(beanDefinition.getBeanClassName()), toAdd);
42          }
43        }
44  
45        // usuwa ze zbioru te elementy, które już są w kontekście
46        for (String beanName : beanFactory.getBeanDefinitionNames()) {
47          BeanDefinition beanDefinition = beanFactory.getBeanDefinition(beanName);
48          if (beanDefinition.getBeanClassName() != null) {
49            toAdd.remove(Class.forName(beanDefinition.getBeanClassName()));
50          }
51        }
52  
53        // tworzy nowe definicje beanów dla brakujących elementów
54        for (Class clazz : toAdd) {
55          factory.registerBeanDefinition(clazz.getSimpleName(), rootBeanDefinition(clazz).getBeanDefinition());
56        }
57      } catch (Exception e) {
58        throw new RuntimeException(e);
59      }
60    }
61  }
