22    default <E extends IPrintFacade<F>, F extends IPrintElementFacade> void addElements(
23        KeyValuePrint print, E facade, Consumer<? super F>... elementInitializations) {
24      for (Consumer<? super F> elementInitialization : elementInitializations) {
25        print.getElements().add(new KeyValuePrintElement());
26        elementInitialization.accept(last(facade.getElements()));
27      }
28    }
29  
30  
31    default <E extends IPrintFacade<F>, F extends IPrintElementFacade> KeyValuePrint initializePrint(
32        Class<E> printClass, Integer idEntity, Consumer<E> initializePrint,
33        Consumer<? super F> ... elementInitializations) {
34      KeyValuePrint print = KeyValuePrint.print(printClass, idEntity).get();
35      getFacadeBuilder().doWithFacade(print, printClass, facade -> {
36        // first add all positions
37        addElements(print, facade, elementInitializations);
38        // then initialize print (may contain summary of positions)
39        initializePrint.accept(facade);
40      });
41      return print;
42    }
43  
44  
45    default <E extends IPrintFacade<F>, F extends IPrintElementFacade> E initializeFacade(
46        Class<E> printClass, Integer idEntity, Consumer<E> initializePrint,
47        Consumer<? super F> ... elementInitializations) {
48      return getFacadeBuilder().createFacade(initializePrint(
49          printClass, idEntity, initializePrint, elementInitializations), printClass);
50    }
51  
52  
53    default <E extends IPrintFacade<F>, F extends IPrintElementFacade, P extends KeyValuePrint> P initializePrint(
54        P print, Class<E> printClass, Consumer<? super E> initializePrint, Consumer<? super F> ... elementInitializations) {
55      getFacadeBuilder().doWithFacade(print, printClass, facade -> {
56        // first add all positions
57        addElements(print, facade, elementInitializations);
58        // then initialize print (may contain summary of positions)

62    }
63  
64  
65    default <E extends IPrintFacade<F>, F extends IPrintElementFacade, P extends KeyValuePrint> E initializeFacade(
66        P print, Class<E> printClass, Consumer<E> initializePrint, Consumer<? super F> ... elementInitializations) {
67      return getFacadeBuilder().createFacade(initializePrint(
68          print, printClass, initializePrint, elementInitializations));
69    }
