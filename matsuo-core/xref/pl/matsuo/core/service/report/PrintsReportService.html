3   import org.springframework.stereotype.Service;
4   import pl.matsuo.core.model.organization.AbstractParty;
5   import pl.matsuo.core.model.organization.Person;
6   import pl.matsuo.core.model.print.KeyValuePrint;
7   import pl.matsuo.core.model.print.initializer.PrintInitializer;
8   import pl.matsuo.core.model.query.Query;
9   import pl.matsuo.core.model.query.QueryBuilder;
10  import pl.matsuo.core.model.query.condition.Condition;
11  import pl.matsuo.core.model.report.IPrintsReportParams;
12  
13  import static pl.matsuo.core.model.query.QueryBuilder.*;
14  
15  
16  @Service
17  public class PrintsReportService extends AbstractReportService<IPrintsReportParams> {
18  
19  
20    /**
21     * Buduje zapytanie do bazy na podstawie przekazanych parametrow
22     * @param params parametry przekazane w requescie
23     * @return Zwraca obiekt zapytania
24     */
25    private static Query<KeyValuePrint> buildQuery(IPrintsReportParams params) {
26      return QueryBuilder.query(KeyValuePrint.class, new Condition[] {
27          maybe(params.getStartDate(), ge("createdTime", params.getStartDate())),
28          maybe(params.getEndDate(), le("createdTime", params.getEndDate())),
29          maybeEq(params.getPatient(), "patient"),
30          maybeEq(params.getPayer(), "payer"),
31          maybeEq(params.getPrintClass(), "printClass")
32      }).initializer(new PrintInitializer());
33    }
34  
35  
36    @Override
37    protected void injectModel(DataModelBuilder dataModel, IPrintsReportParams params) {
38      dataModel.put("prints", database.find(buildQuery(params)))
39          .maybePut(params.getIdPatient(), "patient", Person.class)
40          .maybePut(params.getIdPayer(), "payer", AbstractParty.class);
41    }
42  }
