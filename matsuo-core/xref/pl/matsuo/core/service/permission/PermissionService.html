37    protected String permissionFilePath = "/permissions.json";
38    protected Permissions permissions;

41    protected long interval = 2000;

45    protected Permissions getPermissions() {
46      try {
47        Resource resource = resourceLoader.getResource(permissionFilePath);
48  
49        if (permissions == null
50              || ((lastCheckTime + interval < currentTimeMillis())
51                    && lastReadTime < resource.getFile().lastModified())) {
52          logger.info("reading new permissions from: " + resource.getFile().getAbsolutePath());
53  
54          permissions = new Gson().fromJson(new InputStreamReader(resource.getInputStream()), Permissions.class);
55          lastReadTime = resource.getFile().lastModified();
56          lastCheckTime = currentTimeMillis();
57        }
58      } catch (Exception e) {
59        throw new RuntimeException(e);
60      }
61  
62      return permissions;
63    }

72    @Override
73    public boolean isPermitted(String name, RequestType requestType) {
74      // weryfikacja czasu nieaktywno≈õci - po przekroczeniu automatycznie wylogowywuje
75      if (authorizationLength > 0 && sessionState.getUser() != null
76            && (sessionState.getLastRequestTime() + authorizationLength * 60 * 1000) < currentTimeMillis()) {
77        logoff();
78      }
79  
80      // aktualizacja czasu ostatniego odwo≈Çania do serwisu
81      // fixme: mechanizm wy≈ÇƒÖcze≈Ñ dla cyklicznych zapyta≈Ñ nie bƒôdƒÖcych wynikiem interakcji u≈ºytkownika
82      sessionState.setLastRequestTime(currentTimeMillis());
83  
84      // admin mo≈ºe wszystko
85      if (sessionState.isInGroup(ADMIN.name())) {
86        return true;
87      } else if (requestType.equals(RequestType.OPTIONS)) {
88        // na potrzeby CORS!
89        return true;
90      }
91  
92      // zbi√≥r definicji funkcji pasujƒÖcych do przekazanej nazwy
93      Set<String> functionNames = functionDefinitions(name, requestType);
94      // dodanie nazw zbior√≥w funkcji zawierajƒÖcych definicje matchujƒÖce nazwƒô
95      Set<String> functionDefinitions = functionSets(functionNames, requestType);
96  
97      if (functionDefinitions.isEmpty()) {
98        return false;
99      }
100 
101     // weryfikacja nadania uprawnienia dla danego u≈ºytkownika i danej nazwy funkcji
102     return findPermission(functionDefinitions);
103   }

112   /**
113    * Czy kt√≥ry≈õ ze zbior√≥w funkcji uprawniajƒÖcych do wykonania akcji jest uprawniony dla kt√≥rej≈õ z grup u≈ºytkownika.
114    */
115   protected boolean findPermission(Set<String> elements) {
116     for (String groupName : getPermissions().getPermissions().keySet()) {
117       if (sessionState.isInGroup(groupName)) {
118         List<String> groupPermissions = getPermissions().getPermissions().get(groupName);
119         for (String groupPermission : groupPermissions) {
120           if (elements.contains(groupPermission)) {
121             return true;
122           }
123         }
124       }
125     }
126 
127     return false;
128   }

131   /**
132    * Tworzy zbi√≥r zbior√≥w definicji funkcji, w kt√≥rych znajduje siƒô dowolna z pasujƒÖcych nazw funkcji.
133    */
134   protected Set<String> functionSets(Set<String> functionNames, RequestType requestType) {
135     Set<String> functionSets = new HashSet<>();
136 
137     for (String functionsSetName : getPermissions().getFunctionSets().keySet()) {
138       Map<String, List<String>> functionSet = getPermissions().getFunctionSets().get(functionsSetName);
139       for (String functionName : functionSet.keySet()) {
140         if (functionNames.contains(functionName) && functionSet.get(functionName).contains(requestType.name())) {
141           functionSets.add(functionsSetName);
142         }
143       }
144     }
145 
146     return functionSets;
147   }

150   /**
151    * Czy dana nazwa pasuje do przekazanej funkcji (definicja uprawnienia).
152    */
153   protected boolean matches(String name, String function) {
154     if (function.equals(name)) {
155       return true;
156 
157       // zapis funkcji '/api/xxx/*' spowoduje automatyczne zezwolenie dla '/api/xxx', czyli ka≈ºda nazwa, w≈ÇƒÖcznie z
158       // brakiem slasha i czegokolwiek po xxx jest dozwolona
159     } else if (function.endsWith("*") && name.indexOf(function.substring(0, function.length() - 2)) == 0) {
160       if (function.length() <= 4) {
161         throw new RuntimeException("Wildcard definition must be longer than 4 characters: " + function);
162       }
163 
164       return true;
165     }
166 
167     return false;
168   }

171   /**
172    * Tworzy zbi√≥r definicji funkcji, kt√≥re pasujƒÖ do przekazanego uprawnienia.
173    */
174   protected Set<String> functionDefinitions(String name, RequestType requestType) {
175     Set<String> functionDefinitions = new HashSet<>();
176 
177     for (String function : getPermissions().getFunctions().keySet()) {
178       if (matches(name, function) && getPermissions().getFunctions().get(function).contains(requestType.name())) {
179         functionDefinitions.add(function);
180       }
181     }
182 
183     return functionDefinitions;
184   }
