41    @Override
42    public String login(LoginData loginData) {
43      if (!notEmpty(loginData.getUsername()) || !notEmpty(loginData.getPassword())) {
44        throw new UnauthorizedException();
45      }
46  
47      User user = database.findOne(query(User.class, eq("username", loginData.getUsername()))
48          .initializer(new UserInitializer()));
49      if (user != null && !user.getBlocked()
50          && user.getPassword().equals(passwordHash(loginData.getPassword()))) {
51        user.setLastLoginTime(new Date());
52        user.getGroups().size();
53        database.update(user);
54  
55        sessionState.setUser(user);
56  
57        return loginData.getUsername();
58      } else {
59        // sleep on failure to obfuscate calculation length and slow down next login attempt
60        try {
61          Thread.sleep(1000 + (long) (Math.random() * 1000));
62        } catch (InterruptedException e) {
63          e.printStackTrace();
64        }
65  
66        throw new UnauthorizedException();
67      }
68    }

71    @Override
72    public void activateAccount(String ticket) {
73      List<User> users = database.find(query(User.class, eq("unblockTicket", ticket)));
74  
75      if (users.size() != 1) {
76        throw new RestProcessingException("Unable to identify account");
77      }
78  
79      User user = users.get(0);
80  
81      if (!user.getBlocked()) {
82        throw new RestProcessingException("Account is not blocked");
83      }
84  
85      user.setBlocked(false);
86      user.setUnblockTicket(null);
87      database.update(user);
88    }

91    @Override
92    public String createAccount(CreateAccountData createAccountData) {
93      User user = database.findOne(query(User.class, eq("username", createAccountData.getUsername())));
94      if (user != null) {
95        throw new RestProcessingException("Username already used");
96      }
97  
98      OrganizationUnit organizationUnit = new OrganizationUnit();
99      organizationUnit.setFullName(createAccountData.getCompanyName());
100     organizationUnit.setShortName(createAccountData.getCompanyShortName());
101     organizationUnit.setNip(createAccountData.getCompanyNip());
102     database.create(organizationUnit);
103     Assert.notNull(organizationUnit.getId());
104     // update idBucket for newly created organization
105     organizationUnit.setIdBucket(organizationUnit.getId());
106     database.update(organizationUnit);
107 
108     sessionState.setIdBucket(organizationUnit.getId());
109 
110     Person person = new Person();
111     database.create(person);
112 
113     user = new User();
114     user.setPerson(person);
115     List<Group> groups = database.findAsAdmin(query(Group.class, eq("name", GroupEnum.SUPERVISOR.name())));
116     Assert.isTrue(groups.size() == 1);
117     user.getGroups().add(groups.get(0));
118     user.setUsername(createAccountData.getUsername());
119     user.setPassword(passwordHash(createAccountData.getPassword()));
120     // only a little bit randomness
121     user.setUnblockTicket(RandomStringUtils.randomAlphabetic(16));
122     user.setBlocked(true);
123     user.setIdBucket(organizationUnit.getId());
124     database.create(user);
125 
126     if (extensions != null) {
127       for (ILoginServiceExtension extension : extensions) {
128         extension.createAccount(organizationUnit, user);
129       }
130     }
131 
132     // fixme: send this by email to username
133     return user.getUnblockTicket();
134   }
