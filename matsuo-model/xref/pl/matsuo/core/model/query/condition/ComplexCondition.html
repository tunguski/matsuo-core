9   public class ComplexCondition implements Condition {
10  
11  
12    private String joiner;
13    private final Condition[] conditions;
14  
15  
16    public ComplexCondition(String joiner, Condition ... conditions) {
17      this.joiner = joiner;
18      this.conditions = conditions;
19  
20      if (conditions.length == 0) {
21        throw new IllegalStateException("ComplexCondition must containt one or more internal conditions");
22      }
23    }
24  
25  
26    @Override
27    public String print(AbstractQuery query) {
28      StringBuilder sb = new StringBuilder("(");
29  
30      for (Condition condition : conditions) {
31        sb.append(condition.print(query) + " " + joiner + " ");
32      }
33  
34      // usuwa ostatni joiner
35      sb.delete(sb.length() - joiner.length() -2, sb.length());
36  
37      return sb.toString().trim() + ")";
38    }
39  }
