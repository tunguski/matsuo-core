3   import org.hibernate.type.Type;
4   import pl.matsuo.core.model.interceptor.InterceptorComponent;
5   
6   import java.io.Serializable;
7   import java.util.Date;
8   import java.util.function.Supplier;
9   
10  
11  @InterceptorComponent
12  public class AuditTrailInterceptor extends AbstractEntityInterceptor {
13  
14  
15    Supplier<Object> idUserSupplier = () -> sessionState != null && sessionState.getUser() != null
16        ? sessionState.getUser().getId() : null;
17  
18  
19    @Override
20    public boolean onFlushDirty(Object entity, Serializable id, Object[] currentState,
21                        Object[] previousState, String[] propertyNames, Type[] types) {
22      setValue(currentState, propertyNames, "idUserUpdated", idUserSupplier);
23      setValue(currentState, propertyNames, "updatedTime", () -> new Date());
24      return true;
25    }
26  
27  
28    @Override
29    public boolean onSave(Object entity, Serializable id, Object[] state, String[] propertyNames, Type[] types) {
30      setValue(state, propertyNames, "idUserCreated", idUserSupplier);
31      setValue(state, propertyNames, "createdTime", () -> new Date());
32      return true;
33    }
34  }
