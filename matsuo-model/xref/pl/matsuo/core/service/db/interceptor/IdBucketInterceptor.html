3   import org.hibernate.type.Type;
4   import org.springframework.beans.factory.BeanCreationException;
5   import pl.matsuo.core.model.interceptor.InterceptorComponent;
6   
7   import java.io.Serializable;
8   import java.util.function.Supplier;
9   
10  
11  @InterceptorComponent
12  public class IdBucketInterceptor extends AbstractEntityInterceptor {
13  
14  
15    Supplier<Object> idBucketSupplier = () -> {
16      try {
17        return sessionState != null ? sessionState.getIdBucket() : null;
18      } catch (BeanCreationException e) {
19        return null;
20      }
21    };
22  
23  
24    @Override
25    public boolean onSave(Object entity, Serializable id, Object[] state, String[] propertyNames, Type[] types) {
26      if (getValue(state, propertyNames, "idBucket") == null) {
27        setValue(state, propertyNames, "idBucket", idBucketSupplier);
28        return true;
29      }
30  
31      return false;
32    }
33  }
