8   import java.io.Serializable;

11  import static java.util.Arrays.*;
12  
13  
14  public abstract class AbstractEntityInterceptor extends EmptyInterceptor {
15    private static final long serialVersionUID = 1L;
16  
17  
18    @Autowired
19    protected SessionState sessionState;
20  
21  
22    @Override
23    public boolean onFlushDirty(Object entity, Serializable id, Object[] currentState,
24                        Object[] previousState, String[] propertyNames, Type[] types) {
25      return false;
26    }
27  
28  
29    @Override
30    public boolean onSave(Object entity, Serializable id, Object[] state, String[] propertyNames, Type[] types) {
31      return false;
32    }
33  
34  
35    protected void setValue(Object[] currentState, String[] propertyNames, String propertyToSet,
36                            Supplier<Object> valueProvider) {
37      int index = asList(propertyNames).indexOf(propertyToSet);
38  
39      if (index >= 0 && currentState[index] == null) {
40        currentState[index] = valueProvider.get();
41      }
42    }
43  
44  
45    protected <E> E getValue(Object[] currentState, String[] propertyNames, String propertyToGet) {
46      int index = asList(propertyNames).indexOf(propertyToGet);
47  
48      if (index >= 0) {
49        return (E) currentState[index];
50      } else {
51        throw new IllegalArgumentException("No property " + propertyToGet);
52      }
53    }
54  }
