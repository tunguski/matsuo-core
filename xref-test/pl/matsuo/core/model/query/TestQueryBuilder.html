15    @Test
16    public void testQueries() {
17      // podstawowe zapytani
18      assertEquals("FROM pl.matsuo.core.model.query.TheModel theModel", query(TheModel.class).printQuery());
19      // klauzula select
20      assertEquals("SELECT theModel.id, theModel FROM pl.matsuo.core.model.query.TheModel theModel",
21                      query(TheModel.class, select("theModel.id, theModel")).printQuery());
22      // klauzula where
23      assertEquals("FROM pl.matsuo.core.model.query.TheModel theModel WHERE id = 12",
24                      query(TheModel.class, eq("id", 12)).printQuery());
25      // warunek and
26      assertEquals("FROM pl.matsuo.core.model.query.TheModel theModel WHERE (id = 12 AND id < 200)",
27                      query(TheModel.class, and(eq("id", 12), lt("id", 200))).printQuery());
28      // klauzula order by
29      assertEquals("FROM pl.matsuo.core.model.query.TheModel theModel ORDER BY id DESC",
30                      query(TheModel.class).orderBy("id DESC").printQuery());
31      // theta join
32      assertEquals("FROM pl.matsuo.core.model.query.TheModel theModel , " +
33                       "pl.matsuo.core.model.query.TheModel appointment WHERE (appointment.id = id)",
34                      query(TheModel.class, leftJoin("appointment", TheModel.class, cond("appointment.id = id"))).printQuery());
35    }
