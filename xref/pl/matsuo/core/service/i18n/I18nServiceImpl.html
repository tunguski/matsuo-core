3   import org.springframework.beans.factory.annotation.Autowired;
4   import org.springframework.context.MessageSource;
5   import org.springframework.context.NoSuchMessageException;
6   import org.springframework.stereotype.Service;
7   
8   import java.util.Locale;
9   
10  
11  /**
12   * Created by tunguski on 26.09.13.
13   */
14  @Service
15  public class I18nServiceImpl implements I18nService {
16  
17  
18    @Autowired
19    MessageSource[] messageSources;
20  
21  
22    @Override
23    public String translate(String code, Object ... params) {
24      // FIXME: stworzyÄ‡ localeProvidera
25      Locale locale = Locale.getDefault();
26  
27      while (true) {
28        for (MessageSource messageSource : messageSources) {
29          try {
30            return messageSource.getMessage(code, params, locale);
31          } catch (NoSuchMessageException e) {}
32        }
33  
34        if (locale == null) {
35          return code;
36        } else if (locale.toString().contains("_")) {
37          locale = new Locale(locale.toString().substring(0, locale.toString().lastIndexOf("_")));
38        } else if (!locale.toString().isEmpty()) {
39          locale = null;
40        }
41      }
42    }
43  }
