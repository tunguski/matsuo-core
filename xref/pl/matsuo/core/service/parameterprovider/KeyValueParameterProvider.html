17  public class KeyValueParameterProvider extends AbstractParameterProvider<KeyValueEntity> {
18  
19  
20    public KeyValueParameterProvider(KeyValueEntity underlyingEntity) {
21      super(underlyingEntity);
22    }
23  
24  
25    @Override
26    public Object internalGet(String key, Class<?> expectedClass) {
27      return optional((Object) underlyingEntity.getFields().get(key)).orElseGet(
28          () -> processEx(() -> { return getValue(underlyingEntity, key); }, e -> { return null; }));
29    }
30  
31  
32    @Override
33    public void set(String propertyName, Object value) {
34      if (value == null) {
35        underlyingEntity.getFields().put(propertyName, null);
36      } else {
37        Class<?> parameterType = value.getClass();
38        if (parameterType.equals(Integer.class)) {
39          underlyingEntity.getFields().put(propertyName, value.toString());
40        } else if (parameterType.equals(Double.class)) {
41          underlyingEntity.getFields().put(propertyName, value.toString());
42        } else if (parameterType.equals(String.class)) {
43          underlyingEntity.getFields().put(propertyName, value.toString());
44        } else if (parameterType.equals(Date.class)) {
45          underlyingEntity.getFields().put(propertyName, dateFormat.format(value));
46        } else if (parameterType.equals(BigDecimal.class)) {
47          underlyingEntity.getFields().put(propertyName, value.toString());
48        } else if (parameterType.equals(Boolean.class)) {
49          underlyingEntity.getFields().put(propertyName, value.toString());
50        } else if (Enum.class.isAssignableFrom(parameterType)) {
51          underlyingEntity.getFields().put(propertyName, value.toString());
52        } else {
53          throw new IllegalArgumentException("Unknown property type: " + parameterType.getName());
54        }
55      }
56    }
57  }
