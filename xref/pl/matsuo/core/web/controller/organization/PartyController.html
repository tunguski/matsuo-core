21  @RestController
22  @RequestMapping("/party")
23  public class PartyController {
24  
25  
26    @Autowired
27    protected Database database;
28    @Autowired
29    protected PayerController payerController;
30  
31  
32    @RequestMapping(method = GET)
33    public List<SimpleParty> list(IQueryRequestParams params) {
34      List<SimpleParty> result = new ArrayList<>();
35  
36      List<AbstractParty> parties = payerController.list(params);
37      for (AbstractParty party : parties) {
38        result.add(new SimpleParty(party.getId(), party.getName(), AbstractParty.class));
39      }
40  
41      List<Group> groups = database.find(query(Group.class, ilike("name", params.getQuery())));
42      for (Group group : groups) {
43        result.add(new SimpleParty(group.getId(), group.getName(), Group.class));
44      }
45  
46      return result;
47    }
48  }
