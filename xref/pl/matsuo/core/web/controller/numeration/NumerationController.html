31    @Override
32    protected AbstractQuery<Numeration> listQuery(IEntityQueryRequestParams params, Condition... conditions) {

34    }

36  
37    @RequestMapping(method = POST, consumes = {APPLICATION_JSON_VALUE})
38    @ResponseStatus(CREATED)
39    @Override
40    public HttpEntity<Numeration> create(@RequestBody @Valid Numeration entity,
41                                         @Value("#{request.requestURL}") StringBuffer parentUri) {
42      if (entity.getValue() == null) {
43        entity.setValue(entity.getMinValue());
44      }
45      return super.create(entity, parentUri);
46    }
