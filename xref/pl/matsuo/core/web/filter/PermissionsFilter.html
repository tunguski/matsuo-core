22  @Component("permissionsFilter")
23  public class PermissionsFilter extends AbstractFilter {
24  
25  
26    @Autowired
27    PermissionService permissionService;
28  
29  
30    @Override
31    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
32      throws IOException, ServletException {
33      HttpServletRequest httpRequest = (HttpServletRequest) request;
34      HttpServletResponse httpResponse = (HttpServletResponse) response;
35  
36      // FIXME: weryfikacja uprawnień - jeśli są, zostają przesłane dane
37      if (!permissionService.isPermitted(httpRequest.getRequestURI(),
38                                         IPermissionService.RequestType.valueOf(httpRequest.getMethod()))) {
39        System.out.println("no permissions to: " + httpRequest.getRequestURI());
40  
41        // jeśli brak zalogowanego użytkownika, zwrócony zostaje status UNAUTHORIZED, jeśli użytkownik jest zalogowany,
42        // ale nie ma uprawnień do zasobu, zwrócony zostaje status FORBIDDEN
43        httpResponse.setStatus(sessionState.getUser() == null ? UNAUTHORIZED.value() : FORBIDDEN.value());
44      } else {
45        chain.doFilter(request, response);
46      }
47    }
48  }
