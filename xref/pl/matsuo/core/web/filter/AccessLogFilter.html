12  import javax.servlet.http.HttpServletRequest;
13  import javax.servlet.http.HttpServletResponse;
14  import java.io.IOException;

16

18   * Created by tunguski on 22.12.13.

20  @Component("accessLogFilter")
21  public class AccessLogFilter extends AbstractFilter {
22  
23  
24    @Autowired
25    PermissionService permissionService;
26  
27  
28    @Override
29    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
30      throws IOException, ServletException {
31      HttpServletRequest httpRequest = (HttpServletRequest) request;
32      HttpServletResponse httpResponse = (HttpServletResponse) response;
33  
34      AccessLog accessLog = new AccessLog();
35      accessLog.setIp(httpRequest.getRemoteAddr());
36      accessLog.setRequest(httpRequest.getRequestURI());
37      accessLog.setMethod(httpRequest.getMethod());
38      if (!httpRequest.getParameterMap().keySet().isEmpty()) {
39        accessLog.setParameters(httpRequest.getParameterMap().toString());
40      }
41      if (sessionState.getUser() != null) {
42        accessLog.setIdUser(sessionState.getUser().getId());
43      }
44  
45      database.create(accessLog);
46  
47      try {
48        chain.doFilter(request, response);
49      } finally {
50        accessLog.setStatus(httpResponse.getStatus());
51        try {
52          if (httpResponse.getStatus() >= 300 && httpResponse.getStatus() < 400) {
53            database.delete(accessLog);
54          } else {
55            database.update(accessLog);
56          }
57        } catch (RuntimeException e) {
58          // FIXME: obsługa błędu zapisu statusu
59        }
60      }
61    }
62  }
