26    /**
27     * Tworzy nową listę na bazie przekazanej kolekcji, poprzez pobranie wartości <code>property</code>.
28     * Powtórzenia identycznych wartości zostają usunięte.
29     */
30    public static final <E> List<E> collect(Collection<?> collection, String property) {
31      return (List<E>) collection.stream().map((Object object) -> getValue(object, property)).collect(Collectors.toList());
32    }

35    /**
36     * Tworzy kolekcję wartości na podstawie przekazanej listy kolekcji poprzez pobranie wszystkich wartości z
37     * przekazanych kolekcji.
38     */
39    public static final <E> List<E> flatten(Collection<?> ... collections) {
40      List<E> resultList = new ArrayList<>();
41  
42      for (Collection<?> collection : collections) {
43        resultList.addAll((Collection<? extends E>) collection);
44      }
45  
46      return new ArrayList<>(new LinkedHashSet<>(resultList));
47    }

50    /**
51     * Tworzy mapę w której kluczem jest wartość pola <code>property</code> z obiektu kolekcji
52     * <code>collection</code>, a wartością jest tenże obiekt.
53     */
54    public static final <E, F> Map<E, F> toMap(Collection<F> collection, String property) {
55      Map<E, F> resultMap = new HashMap<E, F>();
56      for (F object : collection) {
57        resultMap.put((E) getValue(object, property), object);
58      }
59  
60      return resultMap;
61    }

68     * Tworzy mapę w której klucze zostają przetworzone
69     */
70    public static final <D, E, F> Map<D, F> reMap(Map<? extends E, F> sourceMap, Function<E, D> mapping) {
71      Map<D, F> resultMap = new HashMap<D, F>();
72      for (E key : sourceMap.keySet()) {
73        resultMap.put(mapping.apply(key), sourceMap.get(key));
74      }
75  
76      return resultMap;
77    }

85    public static final <E> List<E> merge(Collection<E> ... collections) {
86      List<E> result = new ArrayList<>();
87  
88      for (Collection<E> collection : collections) {
89        result.addAll(collection);
90      }

92      return result;

96    public static final <E, F extends Collection<E>> F removeNulls(F collection) {
97      Iterator<E> iterator = collection.iterator();
98      while(iterator.hasNext()) {
99        if (iterator.next() == null) {
100         iterator.remove();
101       }
102     }
103 
104     return collection;
105   }

108   public static final Map<String, String> stringMap(String ... keyValues) {
109     Map<String, String> map = new HashMap<>();
110 
111     for(int i = 0; i < (keyValues.length / 2); i++) {
112       map.put(keyValues[i], keyValues[i+1]);
113     }
114 
115     return map;
116   }

119   /**
120    * Metoda dokonujaca redukcji na kolekcji
121    * @param list redukowana kolekcja
122    * @param reducer reduktor
123    * @param <F> typ elementu w kolekcji
124    * @param <T> typ wartosci zwracanej
125    * @return
126    */
127   public static <F,T> T fold(Collection<? extends F> list, final T startValue, BiFunction<T, F, T> reducer) {
128     T value = startValue;
129 
130     for( F item : list ) {
131       value = reducer.apply(value, item);
132     }
133     return value;
134   }

137   public static Stream<Integer> range(final Integer start, final Integer end) {
138     List<Integer> range = new ArrayList<>();
139 
140     for (int i = start; i < end; i++) {
141       range.add(i);
142     }
143 
144     return range.stream();
145   }
