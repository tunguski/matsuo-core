26    public static <E extends AbstractEntity> AbstractQuery<E> query(Class<E> entityClass, QueryPart ... queryParts) {
27      AbstractQuery<E> query = new AbstractQuery<E>(entityClass);
28      query.parts(queryParts);
29      return query;
30    }

33    /**
34     * Tworzy warunek, który jest dodawany wyłącznie gdy obiekt <code>shouldApply</code> nie jest
35     * <code>null</code>em. Jeśli to napis, to nie może być pusty.
36     */
37    public static Condition maybe(final Object shouldApply, final Condition condition) {
38      return maybe(shouldApply != null
39          && (!(shouldApply instanceof String) || !((String)shouldApply).isEmpty()), condition);
40    }

43    /**
44     * Tworzy warunek, który jest dodawany wyłącznie gdy obiekt <code>shouldApply</code> nie jest
45     * <code>null</code>em. Jeśli to napis, to nie może być pusty.
46     */
47    public static Condition maybeEq(final Object value, final String propertyName) {
48      return maybe(value, eq(propertyName, value));
49    }

53     * Tworzy warunek, który jest dodawany wyłącznie gdy obiekt <code>shouldApply</code> nie jest
54     * <code>null</code>em.
55     */
56    public static Condition maybe(final boolean apply, final Condition condition) {
57      if (apply) {
58        return condition;
59      } else {
60        return null;
61      }
62    }

70    protected static Condition operator(String fieldName, String operator, Object value) {
71      return query -> value == null ? "1 = 1" : fieldName + " " + operator + " " + query.propertyValue(value);

115   public static <E extends Collection<?>> Condition in(final String fieldName, final E value) {
116     return query -> {
117       List<String> properties = new ArrayList<>();
118       for (Object o : value) {
119         if (o != null) {
120           properties.add(query.propertyValue(o));
121         }
122       }
123 
124       return fieldName + " in (" + Joiner.on(", ").join(properties) + ")";
125     };
126   }
