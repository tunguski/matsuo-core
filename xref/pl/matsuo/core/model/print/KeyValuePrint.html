24  @Entity

28  
29    @NotNull
30    private Class<? extends IPrintFacade> printClass;
31    /**
32     * Reference to entity on which print is based.
33     */
34    private Integer idEntity;

39    private List<KeyValuePrintElement> elements = new ArrayList<>();
40  
41  
42    @Override @Transient @JsonIgnore
43    public Class<? extends IPrintFacade> getPrintFacadeClass() {

45    }
46  
47  
48    public static Supplier<? extends KeyValuePrint> print(Class<? extends IPrintFacade> clazz, Integer id) {
49      return () -> printInitializer(clazz, id).apply(new KeyValuePrint());
50    }
51  
52  
53    public static <E extends KeyValuePrint> Function<E, E> printInitializer(Class<? extends IPrintFacade> clazz, Integer id) {
54      return print -> {
55        print.setPrintClass(clazz);
56        print.setIdEntity(id);
57        return print;
58      };
59    }
60  
61

64    }

67    }

70    }

73    }
