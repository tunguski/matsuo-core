19  /**
20   * Nadklasa dla wszystkich obiektów występujących jako podmioty w dowolnym kontekście. Podstawowe
21   * przypadki to firma i osoba prywatna.
22   * @author Marek Romanowski
23   * @since Jul 23, 2013
24   */
25  @Entity
26  @Inheritance(strategy=SINGLE_TABLE)
27  public abstract class AbstractParty extends AbstractEntity {
28  
29  
30    @NIP
31    private String nip;
32    @NotNull @Valid
33    @OneToOne(cascade = ALL)
34    private Address address = new Address();
35  
36  
37    @Transient @JsonIgnore
38    public abstract String getName();
39  
40  
41    // getters
42    public String getNip() {
43      return nip;
44    }
45    public void setNip(String nip) {
46      this.nip = nip != null ? nip.replace("-", "") : null;
47    }
48    public Address getAddress() {
49      return address;
50    }
51    public void setAddress(Address address) {
52      this.address = address;
53    }
54  }
