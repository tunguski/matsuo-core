14  public class CreatePdfResult {
15  
16  
17    private String html;
18    private byte[] pdf;
19    private String pdfString;
20  
21  
22    public CreatePdfResult(String html, byte[] pdf) {
23      this.html = html;
24      this.pdf = pdf;
25    }
26  
27  
28    public String getPdfString() {
29      return optional(pdfString).orElseGet(() -> runtimeEx(() -> {
30        PDDocument document = PDDocument.load(new ByteArrayInputStream(pdf));
31        PDFTextStripper stripper = new PDFTextStripper();
32  
33        pdfString = stripper.getText(document);
34        return pdfString;
35      }));
36    }
37  
38  
39    public String getHtml() {
40      return html;
41    }
42    public byte[] getPdf() {
43      return pdf;
44    }
45  }
