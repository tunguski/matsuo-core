85    private OrganizationUnit createCompany(String fullName, String code, String town, String street, String nip,
86        String regon, String ... patientPesels) {
87      OrganizationUnit organizationUnit = new OrganizationUnit();
88  
89      organizationUnit.setFullName(fullName);
90      organizationUnit.setCode(code);
91      organizationUnit.getAddress().setTown(town);
92      organizationUnit.getAddress().setStreet(street);
93      organizationUnit.setNip(nip);
94      organizationUnit.setRegon(regon);
95  
96      List<Person> employees =
97        patientPesels.length > 0 ? database.find(query(Person.class, in("pesel", patientPesels)))
98          : Collections.<Person>emptyList();
99  
100     organizationUnit.getEmployees().addAll(employees);
101     return database.create(organizationUnit);
102   }
