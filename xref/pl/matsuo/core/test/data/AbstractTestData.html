12  public abstract class AbstractTestData implements IExecuteService {
13  
14  
15    @Autowired
16    protected Database database;
17    @Autowired
18    protected SessionState sessionState;
19  
20  
21    protected void withIdBucket(Integer idBucket, Runnable runnable) {
22      sessionState.setIdBucket(idBucket);
23      try {
24        runnable.run();
25      } finally {
26        sessionState.setIdBucket(null);
27      }
28    }
29  
30  
31    protected void withUser(String userName, Runnable runnable) {
32      User user = database.findOne(query(User.class, eq("username", userName)));
33      sessionState.setUser(user);
34      sessionState.setIdBucket(user.getIdBucket());
35      try {
36        runnable.run();
37      } finally {
38        sessionState.setUser(null);
39        sessionState.setIdBucket(null);
40      }
41    }
42  
43  
44    @Override
45    public String getExecuteServiceName() {
46      return getClass().getName();
47    }
48  }
